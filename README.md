def get_sem_change_cands(distances, k=10, freq_min=1):
    sem_change_cands = (distances
        .query('freq_1 > @freq_min and freq_2 > @freq_min')

        .query('lex.str.len() > 3')
        .nlargest(k, 'dist_sem')

        )
